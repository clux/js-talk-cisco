function User(name) {
	this.name = name;
}

User.prototype.context = function () {
	return this;
};
var ea = new User('eirik');
ea.context(); // { name: 'eirik' }
this; // global object

User.prototype.context = function () {
	return (function () {
		return this;
	})();
};
ea.context(); // global object




User.prototype.context = function () {
	return (function () {
		return this;
	}).call(this);
};
ea.context(); // { name: 'eirik' }

	var that = this;
	return (function () {
	  return that;
	})();
